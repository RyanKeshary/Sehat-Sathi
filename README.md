```typescript
// lib/webrtc/codecOptimizer.ts
export function preferVP9Codec(sdp: string): string {
  const lines = sdp.split('\r\n');
  const mVideoLineIndex = lines.findIndex(l => l.startsWith('m=video'));

  if (mVideoLineIndex === -1) return sdp;

  // Find VP9 payload type from rtpmap lines
  const vp9PayloadType = lines
    .find(l => l.includes('VP9'))
    ?.match(/a=rtpmap:(\d+) VP9/)?.[1];

  if (!vp9PayloadType) return sdp;

  // Reorder payload types to put VP9 first
  const mVideoLine = lines[mVideoLineIndex];
  const parts = mVideoLine.split(' ');
  const payloadTypes = parts.slice(3).filter(p => p !== vp9PayloadType);
  lines[mVideoLineIndex] = [
    ...parts.slice(0, 3),
    vp9PayloadType,
    ...payloadTypes
  ].join(' ');

  // Set VP9 bitrate ceiling at 800kbps for 480p
  const insertIndex = mVideoLineIndex + 1;
  lines.splice(insertIndex, 0, 'b=AS:800');

  return lines.join('\r\n');
}

export function setOpusMaxBitrate(sdp: string, maxBitrate = 128): string {
  return sdp.replace(
    /a=fmtp:(\d+) useinbandfec=1/g,
    `a=fmtp:$1 useinbandfec=1; maxaveragebitrate=${maxBitrate * 1000}`
  );
}
```

Sehat Sathi treats the network as a background optimization вҖ” **never a hard dependency**. Every health record written by the application goes to IndexedDB first, encrypted with AES-256-GCM via the browser's native WebCrypto API, and syncs to the server when connectivity allows.

Patients in connectivity dead zones access their complete medical history in under **50ms** вҖ” faster than any cloud-dependent system could achieve even with full internet.

- вң… Dexie.js typed IndexedDB wrapper вҖ” structured health record storage with full TypeScript types
- вң… AES-256-GCM encryption via WebCrypto API for all PHI at rest вҖ” zero plaintext PHI in IndexedDB
- вң… Service Worker with Cache-First strategy for static assets, Network-First for API calls
- вң… Background Sync API with exponential backoff: 30s вҶ’ 60s вҶ’ 2min вҶ’ 5min вҶ’ 10min
- вң… Three-layer storage architecture: State Cache вҶ’ IndexedDB вҶ’ Encrypted Vault
- вң… CRDT-inspired conflict resolution using timestamp Last-Write-Wins for multi-device sync
- вң… ABDM ABHA ID integration вҖ” 14-digit health identifier linking across providers
- вң… Consent-based ABDM data exchange (M1 identity, M2 provider, M3 user milestones)

- вң… Support for record types: Prescriptions, Lab Reports, Doctor Notes, Vitals, Vaccinations

```typescript
// lib/offline/encryptedVault.ts
const ALGORITHM = 'AES-GCM';
const KEY_LENGTH = 256;

export async function generateEncryptionKey(): Promise<CryptoKey> {
  return await window.crypto.subtle.generateKey(
    { name: ALGORITHM, length: KEY_LENGTH },
    true, // extractable for export/backup
    ['encrypt', 'decrypt']
  );
}

export async function encryptRecord(
  data: object,
  key: CryptoKey
): Promise<{ ciphertext: ArrayBuffer; iv: Uint8Array }> {
  const iv = window.crypto.getRandomValues(new Uint8Array(12));
  const encoded = new TextEncoder().encode(JSON.stringify(data));

  const ciphertext = await window.crypto.subtle.encrypt(
    { name: ALGORITHM, iv },
    key,
    encoded
  );

  return { ciphertext, iv };
}

export async function decryptRecord(
  ciphertext: ArrayBuffer,
  iv: Uint8Array,
  key: CryptoKey
): Promise<object> {
  const decrypted = await window.crypto.subtle.decrypt(
    { name: ALGORITHM, iv },
    key,
    ciphertext
  );

  return JSON.parse(new TextDecoder().decode(decrypted));
}

export async function exportKeyForBackup(key: CryptoKey): Promise<string> {
  const exported = await window.crypto.subtle.exportKey('raw', key);
  return btoa(String.fromCharCode(...new Uint8Array(exported)));
}
```

### Service Worker Strategy

```javascript
// public/sw.js вҖ” Cache strategy implementation
const STATIC_CACHE = 'sehat-sathi-static-v1';
const API_CACHE    = 'sehat-sathi-api-v1';

const STATIC_ASSETS = [
  '/', '/dashboard', '/symptom-checker', '/offline.html',
  '/_next/static/css/app.css',
  '/assets/icons/icon-192.png', '/assets/icons/icon-512.png'
];

// On install: pre-cache all static assets
self.addEventListener('install', event => {
  event.waitUntil(
    caches.open(STATIC_CACHE).then(cache => cache.addAll(STATIC_ASSETS))
  );
  self.skipWaiting();
});

// Fetch strategy: Cache-First for static, Network-First for API
self.addEventListener('fetch', event => {
  const { request } = event;
  const url = new URL(request.url);

  // Static assets: Cache-First
  if (url.pathname.startsWith('/_next/static/') || url.pathname.startsWith('/assets/')) {
    event.respondWith(
      caches.match(request).then(cached => cached || fetch(request))
    );
    return;
  }

  // API routes: Network-First with IndexedDB fallback
  if (url.pathname.startsWith('/api/')) {
    event.respondWith(
      fetch(request)
        .then(response => {
          const clone = response.clone();
          caches.open(API_CACHE).then(cache => cache.put(request, clone));
          return response;
        })
        .catch(() =>
          caches.match(request) ||
          new Response(JSON.stringify({ offline: true, data: [] }),
            { headers: { 'Content-Type': 'application/json' }})
        )
    );
    return;
  }

  // Navigation: App shell fallback
  if (request.mode === 'navigate') {
    event.respondWith(fetch(request).catch(() => caches.match('/offline.html')));
  }
});
```

### Background Sync & Conflict Resolution

```typescript
// lib/offline/syncQueue.ts вҖ” Exponential backoff sync
const BACKOFF_DELAYS = [30_000, 60_000, 120_000, 300_000, 600_000]; // ms

export async function processQueue(): Promise<void> {
  const db = getDatabase();
  const pendingItems = await db.syncQueue.where('status').equals('pending').toArray();

  for (const item of pendingItems) {
    try {
      await fetch(item.endpoint, {
        method: item.method,
        body: JSON.stringify(item.payload),
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${item.token}`
        }
      });
      await db.syncQueue.update(item.id, { status: 'synced' });
    } catch {
      const nextDelay = BACKOFF_DELAYS[Math.min(item.retryCount, 4)];
      await db.syncQueue.update(item.id, {
        retryCount: item.retryCount + 1,
        nextRetryAt: Date.now() + nextDelay,
        status: item.retryCount >= 4 ? 'failed' : 'pending'
      });
    }
  }
}

// CRDT-inspired Last-Write-Wins conflict resolution
export function resolveConflict<T extends { updatedAt: number }>(
  local: T, remote: T
): T {
  return local.updatedAt > remote.updatedAt ? local : remote;
}
```

```typescript
// lib/webrtc/bandwidthMonitor.ts
export async function monitorConnectionQuality(
  pc: RTCPeerConnection,
  onQualityChange: (quality: 'excellent' | 'good' | 'fair' | 'poor') => void
): Promise<() => void> {
  const interval = setInterval(async () => {
    const stats = await pc.getStats();
    let packetsLost = 0, packetsSent = 0, rtt = 0;

    stats.forEach(report => {
      if (report.type === 'outbound-rtp' && report.kind === 'video') {
        packetsLost += report.packetsLost ?? 0;
        packetsSent += report.packetsSent ?? 0;
      }
      if (report.type === 'candidate-pair' && report.state === 'succeeded') {
        rtt = report.currentRoundTripTime ?? 0;
      }
    });

    const lossRate = packetsSent > 0 ? packetsLost / packetsSent : 0;

    if      (lossRate < 0.02 && rtt < 0.15) onQualityChange('excellent');
    else if (lossRate < 0.05 && rtt < 0.30) onQualityChange('good');
    else if (lossRate < 0.10 && rtt < 0.50) onQualityChange('fair');
    else                                     onQualityChange('poor');
  }, 2000);

  return () => clearInterval(interval);
}
```

```typescript
// hooks/useVoiceInput.ts
const LANGUAGE_ASR_CODES: Record<string, string> = {
  hi: 'hi-IN', pa: 'pa-IN', bn: 'bn-IN', mr: 'mr-IN', te: 'te-IN',
  ml: 'ml-IN', as: 'as-IN', gu: 'gu-IN', or: 'or-IN', ur: 'ur-PK',
  kn: 'kn-IN', en: 'en-IN',
};

export function useVoiceInput(language: string) {
  const [transcript, setTranscript] = useState('');
  const [isListening, setIsListening] = useState(false);
  const recognitionRef = useRef<SpeechRecognition | null>(null);

  const startListening = useCallback(() => {
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
    if (!SpeechRecognition) { toast.error('Voice input not supported'); return; }

    recognitionRef.current = new SpeechRecognition();
    recognitionRef.current.lang = LANGUAGE_ASR_CODES[language] ?? 'en-IN';
    recognitionRef.current.continuous = false;
    recognitionRef.current.interimResults = true;

    recognitionRef.current.onresult = (event) => {
      setTranscript(event.results[event.results.length - 1][0].transcript);
    };

    recognitionRef.current.onend = () => setIsListening(false);
    recognitionRef.current.start();
    setIsListening(true);
  }, [language]);

  return {
    transcript,
    isListening,
    startListening,
    stopListening: () => recognitionRef.current?.stop()
  };
}
```

**Step 1 вҖ” Clone the repository**
```bash
git clone https://github.com/yourusername/sehat-sathi.git
cd sehat-sathi
```

**Step 2 вҖ” Install dependencies**
```bash
npm install
# or
yarn install
# or
pnpm install
```

**Step 3 вҖ” Configure environment variables**
```bash
cp .env.example .env.local
# Open .env.local and fill in all required values (see Environment Variables section)
```

**Step 4 вҖ” Set up the database**
```bash
# Push Prisma schema to your PostgreSQL database
npx prisma migrate dev --name init

# Seed with demo data (doctors, specializations, sample records)
npx prisma db seed
```

**Step 5 вҖ” Start the development server**
```bash
npm run dev
```

Open [http://localhost:3000](http://localhost:3000) вҖ” the application is running.

### Development Commands

```bash
# Quality checks (run before committing)
```

### Performance Optimizations Applied

- вң… **Dynamic imports** вҖ” Three.js, GSAP, Lottie, Video Room loaded on demand
- вң… **next/image** вҖ” Automatic WebP/AVIF conversion, lazy loading, blur placeholder
- вң… **next/font** вҖ” Self-hosted, subset loaded (Latin + Devanagari only)
- вң… **React.memo()** вҖ” All pure display components memoized
- вң… **useCallback/useMemo** вҖ” Expensive computations cached
- вң… **react-window** вҖ” Doctor list virtualized (renders only visible cards)
- вң… **will-change: transform** вҖ” Only on actively animating elements
- вң… **prefers-reduced-motion** вҖ” All animations disabled on user preference
- вң… **Three.js disabled on mobile** вҖ” CSS gradient fallback on `innerWidth < 768`
- вң… **SWC minification** вҖ” Faster than Babel, smaller output
- вң… **Brotli compression** вҖ” Vercel auto-applies for all static assets

## рҹӨқ Contributing

> **Sehat Sathi is a mission-driven open-source project.** Every contribution вҖ” whether it's fixing a bug, adding a language translation, improving AI prompts, or optimizing WebRTC codec configuration вҖ” directly impacts the healthcare access of millions of rural Indians.
>
> We welcome contributions from developers, medical professionals, UX researchers, linguists, and anyone who believes healthcare is a fundamental human right.

### Contribution Workflow

```bash
# 1. Fork the repository on GitHub

# 2. Clone your fork
git clone https://github.com/YOUR_USERNAME/sehat-sathi.git
cd sehat-sathi

# 3. Create a feature branch (follow naming convention)
git checkout -b feature/malayalam-voice-support
# or: fix/webrtc-audio-dropout-2g
# or: docs/api-reference-update
# or: perf/hero-animation-bundle-reduction

# 4. Make your changes with proper TypeScript types

# 5. Run all checks
npm run lint && npm run type-check && npm run test

# 6. Commit using Conventional Commits format
git commit -m "feat(symptom-checker): add Malayalam language voice support"
git commit -m "fix(webrtc): resolve audio dropout on 2G networks below 100kbps"
git commit -m "perf(landing): reduce hero animation bundle by 40kb via dynamic import"
git commit -m "docs(api): add curl examples for all core endpoints"

# 7. Push and create a Pull Request
git push origin feature/malayalam-voice-support
```

### Commit Message Convention

```
<type>(<scope>): <description>

Types:   feat | fix | perf | docs | style | refactor | test | chore

Scopes:  symptom-checker | webrtc | records | auth | landing | dashboard
         video | offline | translate | admin | ui | api | db | security
```
